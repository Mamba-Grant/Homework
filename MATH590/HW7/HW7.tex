\section{Problem 1}
Let $\mathbf{A} = 
\begin{bmatrix}
1 & 4 & 7 \\
2 & 5 & 8 \\  
3 & 6 & 10
\end{bmatrix}$. Compute the determinant of $\mathbf{A}$ using the following methods:
\begin{enumerate}[(a)]
    \item Cofactor expansion.
    \item Row reduction. 
    \item Using the "exterior" product $(\mathbf{e}_1 + 2\mathbf{e}_2 + 3\mathbf{e}_3)(4\mathbf{e}_1 + 5\mathbf{e}_2 + 6\mathbf{e}_3)(7\mathbf{e}_1 + 8\mathbf{e}_2 + 10\mathbf{e}_3)$.
    \item By calculating directly the volume of the tetrahedron or parallelepiped formed by the column vectors of $\mathbf{A}$.
\end{enumerate}

\begin{callout}{Solution:}
    \begin{enumerate}
        \item The determinant in terms of cofactors is $a_{11}C_{11}-a_{12}C_{12}+a_{13}C_{13}$.
            \begin{align*}
                \left|\begin{array}{cc} 5 & 8 \\ 6 & 10 \end{array}\right| - 4\left|\begin{array}{cc}  2 & 8 \\ 3 & 10 \end{array}\right| + 7 \left|\begin{array}{cc} 2 & 5 \\ 3 & 6 \end{array}\right| = -3
            \end{align*}

        \item The determinant can be found by row reduction by first row reducing, then computing the determinant in the opposite order. 
            \begin{align*}
                \begin{bmatrix}  1 & 4 & 7 \\ 2 & 5 & 8 \\ 3 & 6 & 10 \end{bmatrix} \to 
                \begin{bmatrix}  1 & 4 & 7 \\ 0 & -3 & -6 \\ 0 & -6 & -12 \end{bmatrix} \to 
                \begin{bmatrix}  1 & 4 & 7 \\ 0 & 1 & 2 \\ 0 & -6 & -12 \end{bmatrix} \to 
                \begin{bmatrix}  1 & 0 & -1 \\ 0 & 1 & 2 \\ 0 & 0 & 0 \end{bmatrix} \\
                = -3~\text{Because from step 3 to 2, \textbf{R2} is multiplied by -3}
            \end{align*}

        \item The determinant can be found using the exterior product, which is defined as: $\alpha \wedge \beta \wedge \gamma = a_1 b_2 c_3 + a_2 b_3 c_1 + a_3 b_1 c_2 - a_1 b_3 c_2 - a_2 b_1 c_3 - a_3 b_2 c_1$. Because the matrix \textbf{A} is constructed from three column vectors, we can apply the wedge product to find the determinant.
            \begin{align*}
                (1)(5)(10)+(2)(6)(7)+(3)(4)(8)-(1)(6)(8)-(2)(4)(10)-(3)(5)(7) \\ = -3
            \end{align*}

        \item The volume of a parallelpiped can be found with the triple product (this is only applicable on a 3x3 matrix): $a \cdot {(b \times c)}$. 
            \begin{align*}
                \begin{bmatrix} 1 \\ 2 \\ 3 \end{bmatrix}
                \cdot 
                \left({\begin{bmatrix} 4 \\ 5 \\ 6 \end{bmatrix} \times
                \begin{bmatrix} 7 \\ 8 \\ 10 \end{bmatrix}}\right)
                = -3
            \end{align*}
    \end{enumerate}
\end{callout}

\section{Problem 2}
Use Cramer's rule to solve the system $\mathbf{A}x = \begin{bmatrix} 1 \\ 1 \\ 1 \end{bmatrix}$ with the matrix $\mathbf{A}$ in Problem 1.

\begin{callout}{Solution:}
    \begin{enumerate}
        \item We have already found the determinant of \textbf{A} to be -3.
        \item Now we can write the solutions $x_1$, $x_2$, and $x_3$ in terms of the determinants:
            \begin{align}
                \frac{\left|\begin{array}{ccc} 1 & 4 & 7 \\ 1 & 5 & 8 \\ 1 & 6 & 10 \end{array}\right|}{-3} = 1 \\
                \frac{\left|\begin{array}{ccc} 1 & 1 & 7 \\ 2 & 1 & 8 \\ 3 & 1 & 10 \end{array}\right|}{-3} = -1 \\
                \frac{\left|\begin{array}{ccc} 1 & 4 & 1 \\ 2 & 5 & 1 \\ 3 & 6 & 1 \end{array}\right|}{-3}  = 0
            \end{align}
    \end{enumerate}
\end{callout}
