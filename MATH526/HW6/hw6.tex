\begin{homeworkProblem}[1]
	Given a continuous uniform distribution with support on the interval $[A, B]$, we want to show that:

	\begin{enumerate}[(a)]
		\item $\mu = \frac{A + B}{2}$
		      \begin{callout}{Solution:}
			      $$\frac{1}{b-a}\int_a^b x ~dx = \frac{a+b}{2}$$
		      \end{callout}
		\item $\sigma^2 = \frac{(B - A)^2}{12}$
		      \begin{callout}{Solution:}
			      \begin{align*}
				      \sigma ^{2} & = \langle x \rangle^2 - \langle x^2 \rangle                            \\
				                  & = \left( \frac{a+b}{2} \right)^{2} - \frac{1}{b-a}\int_{a}^{b} x^2 ~dx \\
				                  & = \frac{(b-a)^{2}}{12}
			      \end{align*}
		      \end{callout}
	\end{enumerate}
\end{homeworkProblem}


\begin{homeworkProblem}[4]
	A bus arrives every 10 minutes at a bus stop. It is assumed that the waiting time for a particular individual is a random variable with a continuous uniform distribution.

	\begin{enumerate}[(a)]
		\item What is the probability that the individual waits more than 7 minutes?
		      \begin{callout}{Solution:}
			      $$\int_{7}^{10} \frac{1}{10} ~dx = \frac{3}{10}$$
		      \end{callout}
		\item What is the probability that the individual waits between 2 and 7 minutes?
		      \begin{callout}{Solution:}
			      $$\int_{2}^{7} \frac{1}{10} ~dx = \frac{1}{2}$$
		      \end{callout}
	\end{enumerate}
\end{homeworkProblem}
