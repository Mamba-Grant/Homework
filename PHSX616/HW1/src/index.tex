\begin{homeworkProblem}
    \textbf{(Problem 1.4)} 
    Find the mean, median, and most probable value of $x$ for the following data (from rolling dice).
    \begin{center}
        \begin{tabular}{cc|cc|cc|cc|cc}
            i & $x_i$ & i & $x_i$ & i & $x_i$ & i & $x_i$ & i & $x_i$ \\
            \hline
            1 & 3 & 6 & 8 & 11 & 12 & 16 & 6 & 21 & 5 \\
            2 & 7 & 7 & 9 & 12 & 8 & 17 & 7 & 22 & 10 \\
            3 & 3 & 8 & 7 & 13 & 6 & 18 & 8 & 23 & 8 \\
            4 & 7 & 9 & 5 & 14 & 6 & 19 & 9 & 24 & 8 \\
            5 & 12 & 10 & 7 & 15 & 7 & 20 & 8 & 25 & 8
        \end{tabular}
    \end{center}

   \begin{callout}{Solution:}
   
       \begin{enumerate}[(1)]
           \item \textbf{Mean:} The mean is given by:
              \begin{align*}
                   \bar{x} &= \frac{\sum x_i}{n} \\ 
                   &= \frac{3+7+3+7+12+8+9+7+5+7+12+8+6+6+7+6+7+8+9+8+5+10+8+8+8}{25} \\ 
                   &= \frac{184}{25} = 7.36
              \end{align*}

           \item \textbf{Median:} The data sorted in order is:
               \[
                   3, 3, 5, 5, 6, 6, 6, 7, 7, 7, 7, 7, 7, 8, 8, 8, 8, 8, 8, 8, 9, 9, 10, 12, 12
               \]
               Since there are 25 numbers (odd count), the median is the middle value, which is the 13th number in this ordered list:
               \[
                   \text{Median} = 7
               \]

           \item \textbf{Mode:} The mode is the most frequently occurring value. From the ordered list, we see that 7 and 8 appear the most (six times each):
               \[
                   \text{Mode} = 7, 8
               \]
               Since we have two modes, the data is bimodal.
       \end{enumerate}

   \end{callout}

\end{homeworkProblem}

\begin{homeworkProblem}
    \textbf{(Problem 1.6)} Calculate the standard deviation from the data of Exercise 1.4
   \begin{callout}{Solution:}

       The standard deviation is given by:
        \[
        \sigma = \sqrt{\frac{\sum (x_i - \bar{x})^2}{n}}
        \]
        Computing each squared deviation:
       \begin{gather*}
        (3 - 7.36)^2 = 19.0596, \quad (7 - 7.36)^2 = 0.1296, \quad (3 - 7.36)^2 = 19.0596, \\ 
           \quad (7 - 7.36)^2 = 0.1296, \quad (12 - 7.36)^2 = 21.3376,
       \end{gather*}
        (similar calculations for all 25 values),
        
        Summing up all squared differences:
        \[
        \sum (x_i - \bar{x})^2 = 113.76
        \]

        Now, dividing by \( n = 24 \):
        \[
        \frac{107.64}{25} = 4.74
        \]

        Taking the square root gives the standard deviation:
        \[
        \sigma = \sqrt{4.3056} = 2.18
        \]
   
   \end{callout}
\end{homeworkProblem}
