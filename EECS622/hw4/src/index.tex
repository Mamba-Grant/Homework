\begin{homeworkProblem}
A lossless matching network was properly constructed to match a source with parameters $V_g$ and $Z_g=10-j 15 \Omega$ to a load $Z_L=40 \Omega$ :


\begin{figure}[h]
  \centering
  \includegraphics[width=0.5\textwidth]{../assets/h4p1f1.png}
\end{figure}

If we disconnect the load from the matching network, we find that we have a new equivalent source (the original source followed by the matching network), with parameters $V_{\text {out }}$ and $Z_{\text {out }}$ :


\begin{figure}[h]
  \centering
  \includegraphics[width=0.5\textwidth]{../assets/h4p1f2.png}
\end{figure}

Likewise, if we disconnect the source from the matching network, we find an input impedance (the matching network followed by the load) of $Z_{\text {in }}$ (turn the page!!):

\begin{figure}[h]
  \centering
  \includegraphics[width=0.5\textwidth]{../assets/h4p1f3.png}
\end{figure}

\begin{enumerate}[1.]
\item Determine the impedance values $Z_{\text {in }}$ and $Z_{\text {out }}$.
    \begin{callout}{Solution:}

        We design a matching network for the express purpose of setting the following relationship,
        $$\begin{cases}
            Z_g = Z_{\text{in}}^* = 10+j15 ~\Omega \\ 
            Z_L = Z_{\text{out}}^* = 40 ~\Omega
        \end{cases}$$


    \end{callout}
\item Say the load $Z_L$ absorbs energy at a rate of 800 mW . Determine then the value $\left|V_g\right|$.
    \begin{callout}{Solution:}

    Because this is a \textit{lossless} matching network, we may exploit the following invariance to determine the magnitude $|V_g|$.

    \begin{align*}
        P_g^{avl} &= P_L^{abs} &&\text{(Lossless)} \\
        \frac{|V_g|^{2}}{8R_g} &= P_L^{abs} &&\text{(Known expression for $P^{avl}$)} \\ 
        |V_g| &= \sqrt{(8 R_g) P_L^{abs}} &&\text{(Algebra)} \\ 
        |V_g| &= 8 \mathrm{~V}
    \end{align*}

    Note: $1 ~\Omega = 1 \frac{V^2}{W}$.

    \end{callout}
\end{enumerate}

\end{homeworkProblem}
